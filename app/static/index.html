function applyFilters(summary) { const filtered = JSON.parse(JSON.stringify(summary)); const statusFilter = filters.status; const glFilter = filters.gl; const srcFilter = filters.src; // Filter mismatched sample if (filtered.mismatched_amount_sample) { filtered.mismatched_amount_sample = filtered.mismatched_amount_sample.filter(r => { const statusOk = statusFilter ? r.status === statusFilter : true; const glOk = glFilter ? String(r.gl_account || "") === String(glFilter) : true; const srcOk = srcFilter ? String(r.source_account || "") === String(srcFilter) : true; return statusOk && glOk && srcOk; }); } // Filter counts for charts if (filtered.status_counts && statusFilter) { filtered.status_counts = filtered.status_counts.filter(s => s.status === statusFilter); } if (filtered.exceptions_by_gl && glFilter) { filtered.exceptions_by_gl = filtered.exceptions_by_gl.filter(g => String(g.gl_bucket || g.gl_account || "") === String(glFilter)); } if (filtered.exceptions_by_src && srcFilter) { filtered.exceptions_by_src = filtered.exceptions_by_src.filter(g => String(g.source_bucket || g.source_account || "") === String(srcFilter)); } // Filter time series if (filtered.status_time_series && statusFilter) { filtered.status_time_series = filtered.status_time_series.filter(r => r.status === statusFilter); } return filtered; }
